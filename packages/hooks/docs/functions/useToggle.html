      function ToggleComponent() {
        const { on, set, reset, toggle } = useToggle();

        return (
          <div>
            <p>State: {on ? 'ON' : 'OFF'}</p>
            <button onClick={() => set(true)}>Set to ON</button>
            <button onClick={toggle}>Toggle</button>
